\newcommand{\compararfigsC}[8]{%

		\centering
		\subfloat[#2\label{#3}]{%
			\includegraphics[width=0.49\columnwidth]{#1}}

		\subfloat[#5\label{#6}]{%
			\includegraphics[width=0.49\columnwidth]{#4}}

}
